Atividade 2 - Estrutura Else

Desenvolva uma aplicação web que demonstre o uso da estrutura else. 

<!DOCTYPE html>
<html>
<body>

<form name="frmelse" onsubmit="procFormulario(); return false;">
    Número: <input type="text" id="num"><br>
    <button type="submit">Verificar</button>
</form>

<div id="resultado"></div>

<script>
function procFormulario(){
    let n = parseInt(document.getElementById('num').value);
    if(n % 2 == 0){
        document.getElementById('resultado').innerHTML = "Número par";
    } else {
        document.getElementById('resultado').innerHTML = "Número ímpar";
    }
}
</script>

</body>
</html>
